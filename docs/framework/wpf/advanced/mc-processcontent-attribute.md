```  
<object  
  xmlns:ignorablePrefix="ignorableUri"  
  xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"  
  mc:Ignorable="ignorablePrefix"...

        [content]  
    </ignorablePrefix:ThisElementCanBeIgnored>  
</object>  
```
