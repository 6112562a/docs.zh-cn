 <span data-ttu-id="03a93-106">以下示例生成 CS0073：</span><span class="sxs-lookup"><span data-stu-id="03a93-106">The following sample generates CS0073:</span></span>  
  
```csharp  
// CS0073.cs  
delegate void del();  
  
class Test  
{  
   public event del MyEvent  
   {  
      add;   // CS0073  
      // try the following lines instead  
      // add  
      // {  
      //    MyEvent += value;  
      // }  
      remove  
      {  
         MyEvent -= value;  
      }  
  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
